Get the API key SECURELY from environment variables // We will set this in the Netlify website UI const apiKey = process.env.GEMINI_API_KEY; if (!apiKey) { return { statusCode: 500, body: JSON.stringify({ message: "Server is missing API key configuration." }), }; } const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`; // 3. This is the same AI logic from our prototype const systemPrompt = `You are an expert e-commerce search assistant.
